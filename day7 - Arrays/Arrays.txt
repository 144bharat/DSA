-----------------------------------------------------Declare-----------------------------------------------------------------------------------------

int [] MyArray; --> MyArray of unknown Size.
int [5] MyArray; --> Now MyArray is of (5)Fixed Size.

-----------------------------------------------------Insert Element in a Array-----------------------------------------------------------------------

for(int i = 0; i < 5; i++)
{
    int Value = i + 1;
    MyArray[i] = Value;
}

-----------------------------------------------------Calculate Size of a Array------------------------------------------------------------------------

int Size = sizeof(MyArray) / sizeof(MyArray[0]);
